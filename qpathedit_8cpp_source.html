135 {
136  this->mode = pathMode;

138  this->currentValidPath.clear();
139  this->edit->clear();
140  switch(pathMode) {
141  case ExistingFile:
142  this->dialog->setAcceptMode(QFileDialog::AcceptOpen);

460 QValidator::State PathValidator::validate(QString &text, int &) const
461 {
462  //check if empty is accepted
463  if(text.isEmpty())
464  return this->allowEmpty ? QValidator::Acceptable : QValidator::Intermediate;
465 
466  //nonexisting parent dir is not possible
467  QFileInfo pathInfo(text);
468  if(!pathInfo.dir().exists())
469  return QValidator::Invalid;
470 
471  //if any file -> always valid as long as dir exists
472  if(this->mode == QPathEdit::AnyFile)
473  return QValidator::Acceptable;
474 
475  //for others: if path exists and is file/dir
476  if(pathInfo.exists()) {
477  if(this->mode == QPathEdit::ExistingFile)//if not a file -> dir -> next parent
478  return pathInfo.isFile() ? QValidator::Acceptable : QValidator::Intermediate;
479  else if(this->mode == QPathEdit::ExistingFolder)//if not a dir -> invalid
480  return pathInfo.isDir() ? QValidator::Acceptable : QValidator::Invalid;
481  } else
482  return QValidator::Intermediate;
483 
484  return QValidator::Acceptable;
485 }
